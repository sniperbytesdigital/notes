interface=wlan0

# Use a DIFFERENT subnet for your hotspot!
# Don't use 192.168.0.x because it conflicts with your internet network
dhcp-range=192.168.4.2,192.168.4.50,255.255.255.0,8h

# Your Pi's IP on the HOTSPOT network
# This should be DIFFERENT from your internet IP
dhcp-option=3,192.168.4.1

# DNS server (your Pi)
dhcp-option=6,192.168.4.1

# REMOVE this line - it's breaking DNS!

# Optional: Forward DNS to Google

server=8.8.4.4
