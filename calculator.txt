package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;
import java.util.Arrays;
import java.util.List;

public class MainActivity extends AppCompatActivity {
    private TextView tvResult;

    private String currentInput = "";
    private double firstNumber = 0.0;
    private String currentOperator = "";
    private boolean shouldResetInput = false;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        tvResult = findViewById(R.id.tvResult);

        // Number buttons
        List<Integer> numberButtons = Arrays.asList(
                R.id.btn0, R.id.btn1, R.id.btn2, R.id.btn3, R.id.btn4,
                R.id.btn5, R.id.btn6, R.id.btn7, R.id.btn8, R.id.btn9
        );

        for (Integer id : numberButtons) {
            Button btn = findViewById(id);
            btn.setOnClickListener(new View.OnClickListener() {
                @Override
                public void onClick(View v) {
                    onNumberClick(v);
                }
            });
        }

        // Operator buttons
        findViewById(R.id.btnAdd).setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                onOperatorClick("+");
            }
        });

        findViewById(R.id.btnSubtract).setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                onOperatorClick("-");
            }
        });

        findViewById(R.id.btnMultiply).setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                onOperatorClick("*");
            }
        });

        findViewById(R.id.btnDivide).setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                onOperatorClick("/");
            }
        });

        // Equals button
        findViewById(R.id.btnEquals).setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                onEqualsClick();
            }
        });

        // Clear button
        findViewById(R.id.btnClear).setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                onClearClick();
            }
        });
    }

    private void onNumberClick(View view) {
        Button button = (Button) view;
        String number = button.getText().toString();

        if (shouldResetInput) {
            currentInput = "";
            shouldResetInput = false;
        }

        currentInput += number; //currentinput=123  firstnumber=789 currentOperator="+"
        tvResult.setText(currentInput);  //shouldResetInput=false
    }

    private void onOperatorClick(String operator) {
        if (!currentInput.isEmpty())
        {
            firstNumber = Double.parseDouble(currentInput);
            currentOperator = operator;
            shouldResetInput = true;
        }
    }

    private void onEqualsClick() {
        if (!currentInput.isEmpty() && !currentOperator.isEmpty()) {
            double secondNumber = Double.parseDouble(currentInput);
            double result = 0.0;

            switch (currentOperator) {
                case "+":
                    result = firstNumber + secondNumber;
                    break;
                case "-":
                    result = firstNumber - secondNumber;
                    break;
                case "*":
                    result = firstNumber * secondNumber;
                    break;
                case "/":
                    if (secondNumber != 0.0) {
                        result = firstNumber / secondNumber;
                    } else {
                        result = 0.0;
                    }
                    break;
            }
            tvResult.setText(String.valueOf((int) result));

            currentInput = tvResult.getText().toString();
            currentOperator = "";
            shouldResetInput = true;
        }
    }

    private void onClearClick() {
        currentInput = "";
        firstNumber = 0.0;
        currentOperator = "";
        shouldResetInput = false;
        tvResult.setText("0");
    }
}
